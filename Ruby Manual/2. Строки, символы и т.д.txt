************************************        # определения
''                                          - можно использовать везде их, кроме интерполяции
""                                          - используются везде, и 100% в интерполяции "#{}"
1..99 == 1...100
#                                           - в конце строки (оставить комментарий для себя)
#{..}                                       - интерполяция строк
[...]                                       - Array
{...}                                       - Hash
условия + {...}                             - скобки, этог do ... end
./                                          - требования require ("./file1.rb") L16 (фаилы аа.rb & bb.rb), означает - что подключает/открывает фаил в текущем каталоге
lambda                                      - (L16.3.1.rb)
yield                                       - добавляется в функцию, для определения (если не добавить, то функцию def test ... end      ->       test {...} ничего не произойдёт)

-----------------                           # математика/сравнения
+=                                          - сложение
==                                          - равно 
!                                           - (не) (not)
!=                                          - не равно  
<                                           - меньше
>                                           - больше  
<=                                          - меньше или равно  
>=                                          - больше или равно
||                                          - символ пайпа, означает (или) (or)
^                                           - Исключающее или xor
&&                                          - и (and)
=>                                          - Hash Rocket


************************************        # переменные
var                                         - локальная переменная, область видимости которой не выходит за пределы блоков/методов где она была определена
$var                                        - глобальная переменная, область видимости - "везде"
@var                                        - переменная экземпляра класса (объекта). Область видимости весь класс.
@@var                                       - переменная класса. Область видимости - весь класс + все экземпляры класса (все объекты).
Const                                       - константа. Пишется всегда с заглавной буквы. Область видимости - "везде". Во многих ЯП (языках программировани) значение константы переопределить нельзя, и это логично. В руби же, менять значение константы можно, зачем - я невкурсе.
:                                           - символ (запомнить объект для любых мест) они для ускорения и оптимизации (L9.1.2.rb)

-----------------                           # аксессоры
attr_reader   - только на чтение            - только getter
attr_writer   - только на изменение/запись  - только setter
attr_accessor - на чтение и изменение       - getter и setter

-----------------                           # три уровня доступа для классов и модулей
public                                      - область видимости в пределах класса, экземпляра класса и всех потомков класса
private                                     - область видимости внутри класса
protected                                   - область видимости внутри класса и всех его потомков

************************************        # работа с фаилами (Указать целый путь, если его не указать, откроется тот что в этой папке где и данный фаил расположен)
File.open("имя.расширение","r")             - "r" - открыть фаил для чтения (L17.4.1.rb)
File.open("имя.расширение","w")             - "w" - открыть фаил для заиси (изменения)
File.open("имя.расширение","w+")            - "w+" или "r+" это для записи и изменения
File.open("имя.расширение","a")             - "a" - append дозаписать фаил (в конец) (L17.4.0.rb)
chmod 777 config.php                        - Change mod (linux)
Dir.chdir                                   - Change dirictory изменить путь фаила

-----------------                           # работа с текстом
>                                           - сохранить в какой либо фаил (при повторном применении, замещает все фаилы!!) ruby L13.1.4.rb > book.txt
>>                                          - сохранить в какой либо фаил, дописав в конце ruby L13.1.4.rb > book.txt
<<                                          - добавление значения в массив
(/\ и т.д.)                                 - чтобы вывести эти символы текстом, надо их дублировать "//"
\n                                          - (new line, line deed) пример ( print "Hello/n" это тоже самое что puts "Hello" )
\t                                          - табуляция (/tname)
\r                                          - (return, carriage return (cr) ) возврат корректки
index                                       - он всегда от 0 и до ... (например в массиве[])
